CREATE OR REPLACE FUNCTION accetta_invito_sicuro(token_invito TEXT)
RETURNS BOOLEAN
LANGUAGE plpgsql
SECURITY DEFINER -- ECCO LA MAGIA: Esegue questa funzione bypassando le RLS
SET search_path = public
AS $$
DECLARE
  v_user_id UUID;
  v_invito_id UUID;
  v_impresa_id UUID;
BEGIN
  -- 1. Chi sta chiamando la funzione?
  v_user_id := auth.uid();
  IF v_user_id IS NULL THEN
    RAISE EXCEPTION 'Devi essere loggato per accettare un invito.';
  END IF;

  -- 2. Cerca il token (Seleziona e blocca la riga per evitare doppie accettazioni simultanee)
  SELECT id, impresa_id INTO v_invito_id, v_impresa_id
  FROM inviti_impresa
  WHERE token = token_invito AND stato = 'in_attesa'
  FOR UPDATE; 

  IF v_invito_id IS NULL THEN
    RAISE EXCEPTION 'Link non valido o già utilizzato.';
  END IF;

  -- 3. Inserisci il dipendente (se esiste già, fallirà grazie ai vincoli del DB)
  INSERT INTO impresa_dipendenti (impresa_id, volontario_id, stato, ruolo_aziendale)
  VALUES (v_impresa_id, v_user_id, 'attivo', 'dipendente');

  -- 4. Brucia il token
  UPDATE inviti_impresa SET stato = 'accettato' WHERE id = v_invito_id;

  RETURN TRUE;
END;
$$;



CREATE OR REPLACE FUNCTION accetta_invito_sicuro(token_invito TEXT)
RETURNS BOOLEAN
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public
AS $$
DECLARE
  v_user_id UUID;
  v_invito_id UUID;
  v_impresa_id UUID;
BEGIN
  v_user_id := auth.uid();
  IF v_user_id IS NULL THEN
    RAISE EXCEPTION 'Devi essere loggato per accettare un invito.';
  END IF;

  -- AGGIUNTO IL CAST ::uuid QUI SOTTO
  SELECT id, impresa_id INTO v_invito_id, v_impresa_id
  FROM inviti_impresa
  WHERE token = token_invito::uuid AND stato = 'in_attesa'
  FOR UPDATE; 

  IF v_invito_id IS NULL THEN
    RAISE EXCEPTION 'Link non valido o già utilizzato.';
  END IF;

  -- Inserimento del dipendente
  INSERT INTO impresa_dipendenti (impresa_id, volontario_id, stato, ruolo_aziendale)
  VALUES (v_impresa_id, v_user_id, 'attivo', 'dipendente');

  -- Brucia il token
  UPDATE inviti_impresa SET stato = 'accettato' WHERE id = v_invito_id;

  RETURN TRUE;
END;
$$;

CREATE OR REPLACE FUNCTION cerca_posizioni_vicine(
  user_lat float, 
  user_lng float, 
  raggio_km float
)
RETURNS SETOF posizioni AS $$
BEGIN
  RETURN QUERY
  SELECT *
  FROM posizioni
  WHERE ST_DWithin(
    coords,
    ST_SetSRID(ST_MakePoint(user_lng, user_lat), 4326)::geography,
    raggio_km * 1000 -- Trasforma km in metri
  )
  ORDER BY coords <-> ST_SetSRID(ST_MakePoint(user_lng, user_lat), 4326)::geography;
END;
$$ LANGUAGE plpgsql;

DROP FUNCTION IF EXISTS cerca_posizioni_vicine(float, float, float);

-- 2. Creiamo la nuova funzione con il cast a testo (ST_AsText)
CREATE FUNCTION cerca_posizioni_vicine(user_lat float, user_lng float, raggio_km float)
RETURNS TABLE (
  id uuid,
  associazione_id uuid,
  titolo text,
  descrizione text,
  tipo text,
  quando text,
  dove text,
  created_at timestamptz,
  data_esatta date,
  ora_inizio time,
  ora_fine time,
  giorni_settimana text[],
  coords text -- Eccolo qui, ora è ufficialmente testo!
)
LANGUAGE plpgsql
AS $$
BEGIN
  RETURN QUERY
  SELECT 
    p.id, 
    p.associazione_id, 
    p.titolo, 
    p.descrizione, 
    p.tipo, 
    p.quando, 
    p.dove, 
    p.created_at, 
    p.data_esatta, 
    p.ora_inizio, 
    p.ora_fine, 
    p.giorni_settimana,

    p.coords,

CREATE OR REPLACE FUNCTION ricerca_avanzata_posizioni(
  min_lat float, min_lng float, max_lat float, max_lng float,
  search_q text DEFAULT NULL,
  filter_tipo text DEFAULT NULL,
  filter_tags uuid[] DEFAULT NULL,
  filter_competenze uuid[] DEFAULT NULL,
  filter_data date DEFAULT NULL,
  filter_giorni text[] DEFAULT NULL
)
RETURNS TABLE (
  id uuid, titolo text, descrizione text, tipo text, dove text, 
  quando text, data_esatta date, ora_inizio time, ora_fine time, 
  coords text, tags jsonb, competenze jsonb
) 
LANGUAGE plpgsql
AS $$
BEGIN
  RETURN QUERY
  SELECT 
    p.id, p.titolo, p.descrizione, p.tipo, p.dove, 
    p.quando, p.data_esatta, p.ora_inizio, p.ora_fine,
    ST_AsText(p.coords) as coords,
    COALESCE(jsonb_agg(DISTINCT t.name) FILTER (WHERE t.id IS NOT NULL), '[]') as tags,
    COALESCE(jsonb_agg(DISTINCT c.name) FILTER (WHERE c.id IS NOT NULL), '[]') as competenze
  FROM posizioni p
  LEFT JOIN posizione_tags pt ON p.id = pt.posizione_id
  LEFT JOIN tags t ON pt.tag_id = t.id
  LEFT JOIN posizione_competenze pc ON p.id = pc.posizione_id
  LEFT JOIN competenze c ON pc.competenza_id = c.id
  WHERE 
    -- 1. Filtro Geografico (Bordi della mappa)
    p.coords && ST_MakeEnvelope(min_lng, min_lat, max_lng, max_lat, 4326)
    
    -- 2. Ricerca Testuale (Nome/Titolo)
    AND (search_q IS NULL OR p.titolo ILIKE '%' || search_q || '%')
    
    -- 3. Filtro Tipo (Una tantum / Ricorrente)
    AND (filter_tipo IS NULL OR p.tipo = filter_tipo)
    
    -- 4. Filtro Data (Per eventi singoli)
    AND (filter_data IS NULL OR p.data_esatta = filter_data)
    
    -- 5. Filtro Giorni (Per ricorrenti)
    AND (filter_giorni IS NULL OR p.giorni_settimana && filter_giorni)

    -- 6. Filtri Relazionali (Tags e Competenze)
    AND (filter_tags IS NULL OR p.id IN (SELECT posizione_id FROM posizione_tags WHERE tag_id = ANY(filter_tags)))
    AND (filter_competenze IS NULL OR p.id IN (SELECT posizione_id FROM posizione_competenze WHERE competenza_id = ANY(filter_competenze)))
    
  GROUP BY p.id;
END;
$$;

CREATE OR REPLACE FUNCTION public.ricerca_avanzata_posizioni(
    min_lat double precision,
    min_lng double precision,
    max_lat double precision,
    max_lng double precision,
    search_q text DEFAULT NULL,
    filter_tipo text DEFAULT NULL,
    filter_tags text[] DEFAULT NULL,
    filter_competenze text[] DEFAULT NULL
)
RETURNS TABLE (
    id uuid,
    associazione_id uuid,
    titolo text,
    descrizione text,
    tipo text,
    dove text,
    ora_inizio time without time zone,
    ora_fine time without time zone,
    quando text,
    data_esatta date,
    giorni_settimana text[],
    created_at timestamp with time zone,
    lat double precision,
    lng double precision,
    tags text[],
    competenze text[],
    immagine_url text -- 🚨 NUOVO CAMPO
)
LANGUAGE plpgsql
AS $$
BEGIN
    RETURN QUERY
    WITH pos_base AS (
        SELECT 
            p.id,
            p.associazione_id,
            p.titolo,
            p.descrizione,
            p.tipo,
            p.dove,
            p.ora_inizio,
            p.ora_fine,
            p.quando,
            p.data_esatta,
            p.giorni_settimana,
            p.created_at,
            ST_Y(p.coords::geometry) as lat,
            ST_X(p.coords::geometry) as lng,
            p.immagine_id -- 🚨 Prendiamo l'ID dell'immagine
        FROM public.posizioni p
        WHERE p.coords IS NOT NULL 
          AND p.coords::geometry && ST_MakeEnvelope(min_lng, min_lat, max_lng, max_lat, 4326)
    ),
    filtered_by_text AS (
        SELECT pb.*
        FROM pos_base pb
        WHERE search_q IS NULL 
           OR pb.titolo ILIKE '%' || search_q || '%' 
           OR pb.descrizione ILIKE '%' || search_q || '%'
    ),
    filtered_by_type AS (
        SELECT ft.*
        FROM filtered_by_text ft
        WHERE filter_tipo IS NULL OR ft.tipo = filter_tipo
    )
    SELECT 
        f.id,
        f.associazione_id,
        f.titolo,
        f.descrizione,
        f.tipo,
        f.dove,
        f.ora_inizio,
        f.ora_fine,
        f.quando,
        f.data_esatta,
        f.giorni_settimana,
        f.created_at,
        f.lat,
        f.lng,
        -- ARRAY DEI TAG
        ARRAY(
            SELECT t.name 
            FROM public.posizione_tags pt 
            JOIN public.tags t ON pt.tag_id = t.id 
            WHERE pt.posizione_id = f.id
        ) as tags,
        -- ARRAY DELLE COMPETENZE
        ARRAY(
            SELECT c.name 
            FROM public.posizione_competenze pc 
            JOIN public.competenze c ON pc.competenza_id = c.id 
            WHERE pc.posizione_id = f.id
        ) as competenze,
        -- 🚨 ESTRAZIONE URL IMMAGINE TRAMITE JOIN
        ma.url as immagine_url
    FROM filtered_by_type f
    LEFT JOIN public.media_associazioni ma ON f.immagine_id = ma.id -- JOIN CON LA LIBRERIA
    WHERE 
        (filter_tags IS NULL OR EXISTS (
            SELECT 1 FROM public.posizione_tags pt
            JOIN public.tags t ON pt.tag_id = t.id
            WHERE pt.posizione_id = f.id AND t.id = ANY(filter_tags::uuid[])
        ))
        AND
        (filter_competenze IS NULL OR EXISTS (
            SELECT 1 FROM public.posizione_competenze pc
            WHERE pc.posizione_id = f.id AND pc.competenza_id = ANY(filter_competenze::uuid[])
        ));
END;
$$;

    lat double precision,
    lng double precision,
    tags text[],
    competenze text[],
    immagine_url text -- 📸 ECCO IL NOSTRO NUOVO CAMPO
)
LANGUAGE plpgsql
AS $$
BEGIN
    RETURN QUERY
    WITH pos_base AS (
        SELECT 
            p.id,
            p.associazione_id,
            p.titolo,
            p.descrizione,
            p.tipo,
            p.dove,
            p.ora_inizio,
            p.ora_fine,
            p.quando,
            p.data_esatta,
            p.giorni_settimana,
            p.created_at,
            ST_Y(p.coords::geometry) as lat,
            ST_X(p.coords::geometry) as lng,
            p.immagine_id -- Recuperiamo il riferimento all'immagine
        FROM public.posizioni p
        WHERE p.coords IS NOT NULL 
          AND p.coords::geometry && ST_MakeEnvelope(min_lng, min_lat, max_lng, max_lat, 4326)
    ),
    filtered_by_text AS (
        SELECT pb.*
        FROM pos_base pb
        WHERE search_q IS NULL 
           OR pb.titolo ILIKE '%' || search_q || '%' 
           OR pb.descrizione ILIKE '%' || search_q || '%'
    ),
    filtered_by_type AS (
        SELECT ft.*
        FROM filtered_by_text ft
        WHERE filter_tipo IS NULL OR ft.tipo = filter_tipo
    )
    SELECT 
        f.id,
        f.associazione_id,
        f.titolo,
        f.descrizione,
        f.tipo,
        f.dove,
        f.ora_inizio,
        f.ora_fine,
        f.quando,
        f.data_esatta,
        f.giorni_settimana,
        f.created_at,
        f.lat,
        f.lng,
        ARRAY(
            SELECT t.name 
            FROM public.posizione_tags pt 
            JOIN public.tags t ON pt.tag_id = t.id 
            WHERE pt.posizione_id = f.id
        ) as tags,
        ARRAY(
            SELECT c.name 
            FROM public.posizione_competenze pc 
            JOIN public.competenze c ON pc.competenza_id = c.id 
            WHERE pc.posizione_id = f.id
        ) as competenze,
        -- INCOLLIAMO L'URL DELL'IMMAGINE DALLA LIBRERIA
        ma.url as immagine_url
    FROM filtered_by_type f
    LEFT JOIN public.media_associazioni ma ON f.immagine_id = ma.id
    WHERE 
        (filter_tags IS NULL OR EXISTS (
            SELECT 1 FROM public.posizione_tags pt
            JOIN public.tags t ON pt.tag_id = t.id
            WHERE pt.posizione_id = f.id AND t.id = ANY(filter_tags::uuid[])
        ))
        AND
        (filter_competenze IS NULL OR EXISTS (
            SELECT 1 FROM public.posizione_competenze pc
            WHERE pc.posizione_id = f.id AND pc.competenza_id = ANY(filter_competenze::uuid[])
        ));
END;
$$;

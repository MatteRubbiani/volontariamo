-- 1. Crea la tabella profili (L'Hub Centrale)
CREATE TABLE public.profili (
  id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  ruolo TEXT NOT NULL CHECK (ruolo IN ('volontario', 'associazione', 'impresa')),
  created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL,
  
  -- Imposta l'ID come chiave primaria
  PRIMARY KEY (id)
);

-- 2. Abilita la Sicurezza (Row Level Security)
ALTER TABLE public.profili ENABLE ROW LEVEL SECURITY;

-- 3. Crea le Policy di Sicurezza
-- Permette a un utente loggato di leggere SOLO il proprio ruolo
CREATE POLICY "Gli utenti possono vedere il proprio profilo"
  ON public.profili FOR SELECT
  USING ( auth.uid() = id );

-- Permette alla Server Action di inserire il ruolo durante l'onboarding
CREATE POLICY "Gli utenti possono inserire il proprio profilo"
  ON public.profili FOR INSERT
  WITH CHECK ( auth.uid() = id );

-- (Opzionale) Permette all'utente di cambiare ruolo in futuro
CREATE POLICY "Gli utenti possono aggiornare il proprio profilo"
  ON public.profili FOR UPDATE
  USING ( auth.uid() = id );

CREATE OR REPLACE FUNCTION accetta_invito_sicuro(token_invito TEXT)
RETURNS BOOLEAN
LANGUAGE plpgsql
SECURITY DEFINER -- ECCO LA MAGIA: Esegue questa funzione bypassando le RLS
SET search_path = public
AS $$
DECLARE
  v_user_id UUID;
  v_invito_id UUID;
  v_impresa_id UUID;
BEGIN
  -- 1. Chi sta chiamando la funzione?
  v_user_id := auth.uid();
  IF v_user_id IS NULL THEN
    RAISE EXCEPTION 'Devi essere loggato per accettare un invito.';
  END IF;

  -- 2. Cerca il token (Seleziona e blocca la riga per evitare doppie accettazioni simultanee)
  SELECT id, impresa_id INTO v_invito_id, v_impresa_id
  FROM inviti_impresa
  WHERE token = token_invito AND stato = 'in_attesa'
  FOR UPDATE; 

  IF v_invito_id IS NULL THEN
    RAISE EXCEPTION 'Link non valido o già utilizzato.';
  END IF;

  -- 3. Inserisci il dipendente (se esiste già, fallirà grazie ai vincoli del DB)
  INSERT INTO impresa_dipendenti (impresa_id, volontario_id, stato, ruolo_aziendale)
  VALUES (v_impresa_id, v_user_id, 'attivo', 'dipendente');

  -- 4. Brucia il token
  UPDATE inviti_impresa SET stato = 'accettato' WHERE id = v_invito_id;

  RETURN TRUE;
END;
$$;



CREATE OR REPLACE FUNCTION accetta_invito_sicuro(token_invito TEXT)
RETURNS BOOLEAN
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public
AS $$
DECLARE
  v_user_id UUID;
  v_invito_id UUID;
  v_impresa_id UUID;
BEGIN
  v_user_id := auth.uid();
  IF v_user_id IS NULL THEN
    RAISE EXCEPTION 'Devi essere loggato per accettare un invito.';
  END IF;

  -- AGGIUNTO IL CAST ::uuid QUI SOTTO
  SELECT id, impresa_id INTO v_invito_id, v_impresa_id
  FROM inviti_impresa
  WHERE token = token_invito::uuid AND stato = 'in_attesa'
  FOR UPDATE; 

  IF v_invito_id IS NULL THEN
    RAISE EXCEPTION 'Link non valido o già utilizzato.';
  END IF;

  -- Inserimento del dipendente
  INSERT INTO impresa_dipendenti (impresa_id, volontario_id, stato, ruolo_aziendale)
  VALUES (v_impresa_id, v_user_id, 'attivo', 'dipendente');

  -- Brucia il token
  UPDATE inviti_impresa SET stato = 'accettato' WHERE id = v_invito_id;

  RETURN TRUE;
END;
$$;

CREATE OR REPLACE FUNCTION cerca_posizioni_vicine(
  user_lat float, 
  user_lng float, 
  raggio_km float
)
RETURNS SETOF posizioni AS $$
BEGIN
  RETURN QUERY
  SELECT *
  FROM posizioni
  WHERE ST_DWithin(
    coords,
    ST_SetSRID(ST_MakePoint(user_lng, user_lat), 4326)::geography,
    raggio_km * 1000 -- Trasforma km in metri
  )
  ORDER BY coords <-> ST_SetSRID(ST_MakePoint(user_lng, user_lat), 4326)::geography;
END;
$$ LANGUAGE plpgsql;



-- 1. Facciamo fuori la vecchia funzione
DROP FUNCTION IF EXISTS cerca_posizioni_vicine(float, float, float);

-- 2. Creiamo la nuova funzione con il cast a testo (ST_AsText)
CREATE FUNCTION cerca_posizioni_vicine(user_lat float, user_lng float, raggio_km float)
RETURNS TABLE (
  id uuid,
  associazione_id uuid,
  titolo text,
  descrizione text,
  tipo text,
  quando text,
  dove text,
  created_at timestamptz,
  data_esatta date,
  ora_inizio time,
  ora_fine time,
  giorni_settimana text[],
  coords text -- Eccolo qui, ora è ufficialmente testo!
)
LANGUAGE plpgsql
AS $$
BEGIN
  RETURN QUERY
  SELECT 
    p.id, 
    p.associazione_id, 
    p.titolo, 
    p.descrizione, 
    p.tipo, 
    p.quando, 
    p.dove, 
    p.created_at, 
    p.data_esatta, 
    p.ora_inizio, 
    p.ora_fine, 
    p.giorni_settimana,
    ST_AsText(p.coords) as coords -- La magia della traduzione
  FROM posizioni p
  WHERE p.coords IS NOT NULL
  AND ST_DWithin(
    p.coords, 
    ST_SetSRID(ST_MakePoint(user_lng, user_lat), 4326), 
    raggio_km * 1000 -- PostGIS lavora in metri, quindi moltiplichiamo
  );
END;
$$;

CREATE OR REPLACE FUNCTION ricerca_avanzata_posizioni(
  min_lat float, min_lng float, max_lat float, max_lng float,
  search_q text DEFAULT NULL,
  filter_tipo text DEFAULT NULL,
  filter_tags uuid[] DEFAULT NULL,
  filter_competenze uuid[] DEFAULT NULL,
  filter_data date DEFAULT NULL,
  filter_giorni text[] DEFAULT NULL
)
RETURNS TABLE (
  id uuid, titolo text, descrizione text, tipo text, dove text, 
  quando text, data_esatta date, ora_inizio time, ora_fine time, 
  coords text, tags jsonb, competenze jsonb
) 
LANGUAGE plpgsql
AS $$
BEGIN
  RETURN QUERY
  SELECT 
    p.id, p.titolo, p.descrizione, p.tipo, p.dove, 
    p.quando, p.data_esatta, p.ora_inizio, p.ora_fine,
    ST_AsText(p.coords) as coords,
    COALESCE(jsonb_agg(DISTINCT t.name) FILTER (WHERE t.id IS NOT NULL), '[]') as tags,
    COALESCE(jsonb_agg(DISTINCT c.name) FILTER (WHERE c.id IS NOT NULL), '[]') as competenze
  FROM posizioni p
  LEFT JOIN posizione_tags pt ON p.id = pt.posizione_id
  LEFT JOIN tags t ON pt.tag_id = t.id
  LEFT JOIN posizione_competenze pc ON p.id = pc.posizione_id
  LEFT JOIN competenze c ON pc.competenza_id = c.id
  WHERE 
    -- 1. Filtro Geografico (Bordi della mappa)
    p.coords && ST_MakeEnvelope(min_lng, min_lat, max_lng, max_lat, 4326)
    
    -- 2. Ricerca Testuale (Nome/Titolo)
    AND (search_q IS NULL OR p.titolo ILIKE '%' || search_q || '%')
    
    -- 3. Filtro Tipo (Una tantum / Ricorrente)
    AND (filter_tipo IS NULL OR p.tipo = filter_tipo)
    
    -- 4. Filtro Data (Per eventi singoli)
    AND (filter_data IS NULL OR p.data_esatta = filter_data)
    
    -- 5. Filtro Giorni (Per ricorrenti)
    AND (filter_giorni IS NULL OR p.giorni_settimana && filter_giorni)

    -- 6. Filtri Relazionali (Tags e Competenze)
    AND (filter_tags IS NULL OR p.id IN (SELECT posizione_id FROM posizione_tags WHERE tag_id = ANY(filter_tags)))
    AND (filter_competenze IS NULL OR p.id IN (SELECT posizione_id FROM posizione_competenze WHERE competenza_id = ANY(filter_competenze)))
    
  GROUP BY p.id;
END;
$$;



CREATE OR REPLACE FUNCTION ricerca_avanzata_posizioni(
  min_lat float, min_lng float, max_lat float, max_lng float,
  search_q text DEFAULT NULL,
  filter_tipo text DEFAULT NULL,
  filter_tags uuid[] DEFAULT NULL,
  filter_competenze uuid[] DEFAULT NULL,
  filter_data date DEFAULT NULL,
  filter_giorni text[] DEFAULT NULL
)
RETURNS TABLE (
  id uuid, titolo text, descrizione text, tipo text, dove text, 
  quando text, data_esatta date, ora_inizio time, ora_fine time, 
  coords text, tags jsonb, competenze jsonb
) 
LANGUAGE plpgsql
AS $$
BEGIN
  RETURN QUERY
  SELECT 
    p.id, p.titolo, p.descrizione, p.tipo, p.dove, 
    p.quando, p.data_esatta, p.ora_inizio, p.ora_fine,
    ST_AsText(p.coords) as coords,
    COALESCE(jsonb_agg(DISTINCT t.name) FILTER (WHERE t.id IS NOT NULL), '[]') as tags,
    COALESCE(jsonb_agg(DISTINCT c.name) FILTER (WHERE c.id IS NOT NULL), '[]') as competenze
  FROM posizioni p
  LEFT JOIN posizione_tags pt ON p.id = pt.posizione_id
  LEFT JOIN tags t ON pt.tag_id = t.id
  LEFT JOIN posizione_competenze pc ON p.id = pc.posizione_id
  LEFT JOIN competenze c ON pc.competenza_id = c.id
  WHERE 
    -- 🚨 LA CORREZIONE È QUI: cast p.coords::geometry
    p.coords::geometry && ST_MakeEnvelope(min_lng, min_lat, max_lng, max_lat, 4326)
    
    AND (search_q IS NULL OR search_q = '' OR p.titolo ILIKE '%' || search_q || '%')
    AND (filter_tipo IS NULL OR filter_tipo = '' OR p.tipo = filter_tipo)
    AND (filter_data IS NULL OR p.data_esatta = filter_data)
    AND (filter_giorni IS NULL OR p.giorni_settimana && filter_giorni)
    AND (filter_tags IS NULL OR p.id IN (SELECT posizione_id FROM posizione_tags WHERE tag_id = ANY(filter_tags)))
    AND (filter_competenze IS NULL OR p.id IN (SELECT posizione_id FROM posizione_competenze WHERE competenza_id = ANY(filter_competenze)))
  GROUP BY p.id;
END;
$$;
